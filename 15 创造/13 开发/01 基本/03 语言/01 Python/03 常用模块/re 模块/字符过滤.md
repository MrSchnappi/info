```
def remove_punctuation(line):
    """
    去除所有半角全角符号，只留字母、数字、中文
    """
    rule = re.compile(r"[^a-zA-Z0-9\u4e00-\u9fa5]")
    line = rule.sub('', line)
    return line

def check_punctuation(word):
    pattern = re.compile(r"[^a-zA-Z0-9\u4e00-\u9fa5]")
    if pattern.search(word):
        return True
    else:
        return False
```
